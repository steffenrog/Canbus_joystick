81async def send_joystick_position(x, y, button_values):
82 """
83 Info: This function sends the joystick position, button values over the CAN bus.
84
85 Args:
86 x (int): Joystick x-axis value.
87 y (int): Joystick y-axis value.
88 button_values (list): List of button values, each value is either True or False.
89
90 Parameters:
91 id (int): The CAN message identifier.
92 joy_res (int): Resolution of the joystick, default is 12. Options are 8, 10, and 12.
93
94 Returns: None
95
96 Return type: None
97 """
98 id = 0x18fdd6F1 #ID mimics Grayhill
99 joy_res = 12 #Default value 12, options 8, 10, 12.
100
101
102 x = ((x/2**16)*(2**(joy_res+1)))-(2**joy_res) #Works pretty good
103 y = ((y/2**16)*(2**(joy_res+1)))-(2**joy_res) #Works pretty good
104
105
106 data = [0x01, 0x00, 0x01, 0x00, 0xff, 0x00, 0x00, 0x1f]
107
108 if button_values[0]:
109 data[5] = data[5] | 0x40
110 if button_values[1]:
111 data[5] = data[5] | 0x10
112 if button_values[2]:
113 data[5] = data[5] | 0x04
114 if button_values[3]:
115 data[5] = data[5] | 0x01
116 if button_values[4]:
117 data[6] = data[6] | 0x04
118 if button_values[5]:
119 data[6] = data[6] | 0x40
120
121
122 if joy_res == 12:
123 data[0] = 0x03
124 data[2] = 0x03
125 elif joy_res == 10:
126 data[0] = 0x03
127 data[2] = 0x00
128 else:
129 data[0] = 0x00
130 data[2] = 0x00
131 if x < 0: data[0] = data[0] | 0x10
132 if y < 0: data[2] = data[2] | 0x10
133
134 tmp = int(abs(x)).to_bytes(2,'big', signed=False)
135 data[0] = data[0] | ((tmp[0] << 6) & 0xC0) # 10 bit ...
136 data[0] = data[0] | (tmp[0] & 0x0c) # 12 bit ...
137 data[1] = tmp[1] # 8 bit ...
138
139 tmp = int(abs(y)).to_bytes(2,'big', signed=False)
140 data[2] = data[2] | ((tmp[0] << 6) & 0xC0)
141 data[2] = data[2] | (tmp[0] & 0x0c)
142 data[3] = tmp[1]
143
144 #print(bytes(data)) #Debugging print
145
146
147 message = Message(id=id, data=bytes(data), extended=True)
148 can_bus.send(message) #Comment line to run without can.
149 await asyncio.sleep(0.05)

202async def listen_can(listener):
203 """
204 Info: This function listens for CAN messages and updates the status of LEDs based on the received data.
205
206 Args:
207 listener (object): An object representing a CAN listener.
208
209 Parameters:
210 message_count (int): The number of incoming CAN messages.
211 msg (object): A received CAN message.
212 k (int): The key received from the other part of the receiving software.
213 c (int): The center LED status received from the other part of the receiving software.
214 led_status (list): A list of LED status values.
215
216 Returns:
217 None.
218
219 Return type:
220 None.
221 """
222 while True:
223 message_count = listener.in_waiting()
224 for _i in range(message_count):
225 msg = listener.receive()
226
227 #print(msg.id,msg.data) #Debugging print
228
229 if msg.id == 0x18eff102:
230 k = msg.data[0]
231 c = int((msg.data[1]&0xF0) >> 4)
232 led_status[k-1].value= (c==0) #OK
233
234 await asyncio.sleep(0)
