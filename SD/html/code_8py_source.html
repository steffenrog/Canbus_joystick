139async def send_joystick_position(x, y):
140 """
141 Description:
142 This function sends the joystick position by constructing a CAN message and sending it through the CAN bus.
143
144 Args:
145 x (int): The x-coordinate of the joystick position.
146 y (int): The y-coordinate of the joystick position.
147
148 Attributes:
149 id (int): The ID of the CAN message.
150 data (list): The data array that forms the content of the CAN message.
151 message (Message): The constructed CAN message.
152
153 Returns:
154 None
155
156 Modifies:
157 Sends a CAN message through the CAN bus.
158 """
159 id = 0x18fdd6F1 #ID mimics Grayhill
160
161
162 x = ((x/2**16)*(2**(joy_res+1)))-(2**joy_res) #Works pretty good
163 y = ((y/2**16)*(2**(joy_res+1)))-(2**joy_res) #Works pretty good
164
165 #print(x,y) #Debugging print
166
167
168 data = [0x01, 0x00, 0x01, 0x00, 0xff, 0x00, 0x00, 0x1f]
169
170
171 if not buttons[0]:
172 data[5] = data[5] | 0x40
173 if not buttons[1]:
174 data[5] = data[5] | 0x10
175 if not buttons[2]:
176 data[5] = data[5] | 0x04
177 if not buttons[3]:
178 data[5] = data[5] | 0x01
179 if not buttons[4]:
180 data[6] = data[6] | 0x04
181 if not buttons[5]:
182 data[6] = data[6] | 0x40
183
184 #print(btn1.value, btn2.value, btn3.value, btn4.value) #Debug print
185
186
187 if joy_res == 12:
188 data[0] = 0x03
189 data[2] = 0x03
190 elif joy_res == 10:
191 data[0] = 0x03
192 data[2] = 0x00
193 else:
194 data[0] = 0x00
195 data[2] = 0x00
196
197 if x < 0: data[0] = data[0] | 0x10
198 if y < 0: data[2] = data[2] | 0x10
199
200 tmp = int(abs(x)).to_bytes(2,'big', signed=False)
201 data[0] = data[0] | ((tmp[0] << 6) & 0xC0) # 10 bit ...
202 data[0] = data[0] | (tmp[0] & 0x0c) # 12 bit ...
203 data[1] = tmp[1] # 8 bit ...
204 tmp = int(abs(y)).to_bytes(2,'big', signed=False)
205 data[2] = data[2] | ((tmp[0] << 6) & 0xC0)
206 data[2] = data[2] | (tmp[0] & 0x0c)
207 data[3] = tmp[1]
208
209 print(bytes(data)) #Debugging print
210
211 #Send canmessage, buttons and joystick
212 message = Message(id=id, data=bytes(data), extended=True)
213 can_bus.send(message) #Comment line to run without can.
214 await asyncio.sleep(0.1)

267async def listen_can(listener):
268 """
269 Description:
270 This code runs an asynchronous function to listen on the CAN bus for incoming messages.
271
272 Args:
273 listener: A listener object to listen on the CAN bus.
274
275 Attributes:
276 message_count: An integer representing the number of incoming messages.
277 msg: An incoming message from the CAN bus.

279 c: An integer representing the status of the LED.
280
281 Returns:
282 None
283
284 Modifies:
285 Updates the LED status in the 'led_status' list based on the incoming CAN bus messages.
286 """
287 while True:
288 message_count = listener.in_waiting()
289 for _i in range(message_count):
290 msg = listener.receive()

292 #print("Message from: ", hex(msg.id)) #Debugging print
293 #print(msg.data)
294
295 if msg.id == 0x18eff102:
296 k = msg.data[0]
297 c = int((msg.data[1]&0xF0) >> 4)
298 led_status[k-1].value= (c==0) #OK
